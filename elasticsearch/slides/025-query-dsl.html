Query DSL

curl -XPOST 'http://localhost:9200/cars/_search' -d '
{
  query: {
    match: {  
      text: {
        query: "Ford 2010", operator: "AND"
      }
}}}'

# ... returns records with ford AND 2010 in 'text' field


curl -XPOST 'http://localhost:9200/cars/_search' -d '
{
  query: {
    match_all: {}
}}'

# ... returns all records
