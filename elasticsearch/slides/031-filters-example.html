filters

nested dsl example

curl -XPOST 'http://localhost:9200/cars/_search' -d '
{
  filter: {
    or: [
      {
        and: [
          { term: { model: "kuga" }},
          { term: { year: 2010 }}
  	    ]
      }, {
        and: [
          { term: { manufacturer: "bmw" } },
          { range: { 
            year: { gte: 2010 }
          }}
        ]
      }
    ]
  }
}'
