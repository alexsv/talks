search api endpoints


# search for "Ford" in `cars` index
curl -XGET `http://localhost:9200/cars/_search?q=ford

# search for "Ford" in ALL indexes all fields
curl -XGET `http://localhost:9200/_search?q=ford

# search for "Ford" in `cars` index and `sedan` doc type
curl -XGET `http://localhost:9200/cars/sedan/_search?q=ford

# search for "Ford" in `cars` index and `sedan`,`suv` document types
curl -XGET `http://localhost:9200/cars/sedan,suv/_search?q=ford

... or POST request on _search endpoint
